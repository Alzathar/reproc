14 // Uses the reproc C++ API to print CMake's help page.
15 int main()
16 {
17  // We can pass parameters to the constructor that will be passed to
18  // process::stop in the destructor. cmake --help is a short lived command and
19  // will always exit on its own so waiting indefinitely (reproc::infinite)
20  // until it exits on its own (reproc::cleanup::wait) is sufficient to make
21  // sure the process is stopped and cleaned up.
22  reproc::process cmake_help(reproc::cleanup::wait, reproc::infinite);
23 
24  std::vector<std::string> args = { "cmake", "--help" };
25 
26  std::error_code ec = cmake_help.start(args);
27  if (ec == reproc::errc::file_not_found) {
28  std::cerr << "cmake not found. Make sure it's available from the PATH.";
29  return 1;
30  } else if (ec) {
31  return fail(ec);
32  }
33 
34  std::string output;
35 
36  // string_parser reads the child process output into the given string.
37  ec = cmake_help.read(reproc::stream::out, reproc::string_parser(output));
38  if (ec) { return fail(ec); }
39 
40  std::cout << output << std::flush;
41 
42  // You can also pass an ostream_parser to write the output directly to an
43  // output stream such as std::cerr.
44  ec = cmake_help.read(reproc::stream::err, reproc::ostream_parser(std::cerr));
45  if (ec) { return fail(ec); }
46 
47  // Even if we pass cleanup parameters to the constructor, we can still call
48  // stop ourselves. This allows us to retrieve the exit status of the process
49  // which is hard to get if the process is stopped in the destructor. The
50  // process class has logic to prevent stop from being called in the destructor
51  // if it has already been called by the user.
52  unsigned int exit_status = 0;
53  ec = cmake_help.stop(reproc::cleanup::wait, reproc::infinite, &exit_status);
54  if (ec) { return fail(ec); }
55 
56  return static_cast<int>(exit_status);
57 }
