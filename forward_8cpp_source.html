14 /*
15 Forwards the program arguments to a child process and prints its output on
16 stdout.
17 
18 Example: "./forward cmake --help" will print CMake's help output.
19 
20 This program can be used to verify that manually executing a command and
21 executing it with reproc give the same output.
22 */
23 int main(int argc, char *argv[])
24 {
25  if (argc <= 1) {
26  std::cerr << "No arguments provided. Example usage: ./forward cmake --help";
27  return 1;
28  }
29 
30  // The destructor calls process::stop with the parameters we pass in the
31  // constructor which saves us from having to make sure the process is always
32  // stopped correctly.
33  //
34  // Any kind of process can be started with forward so we make sure the process
35  // is cleaned up correctly by adding the reproc::cleanup::terminate flag
36  // (sends SIGTERM (POSIX) or CTRL-BREAK (Windows) and waits 5 seconds). We
37  // also add the reproc::cleanup::kill flag which sends SIGKILL (POSIX) or
38  // calls TerminateProcess (Windows) if the process hasn't exited after 5
39  // seconds and waits 5 more seconds.
40  //
41  // Note that the timout value of 5s is a maximum time to wait. If the process
42  // exits earlier than the timeout the destructor will return immediately.
43  reproc::process forward(reproc::cleanup::terminate | reproc::cleanup::kill,
44  5000);
45 
46  std::error_code ec = forward.start(argc - 1, argv + 1);
47  if (ec == reproc::errc::file_not_found) {
48  std::cerr << "Program not found. Make sure it's available from the PATH";
49  return 1;
50  } else if (ec) {
51  return fail(ec);
52  }
53 
54  // Some programs wait for the input stream to be closed before continuing so
55  // we close it explicitly.
56  forward.close(reproc::stream::in);
57 
58  // Pipe child process stdout output to std::cout of parent process
59  ec = forward.read(reproc::stream::out, reproc::ostream_parser(std::cout));
60  if (ec) { return fail(ec); }
61 
62  // Pipe child process stderr output to std::cerr of parent process
63  ec = forward.read(reproc::stream::err, reproc::ostream_parser(std::cerr));
64  if (ec) { return fail(ec); }
65 
66  // Call stop ourselves to get the exit_status. See the cmake-help example for
67  // more info.
68  unsigned int exit_status = 0;
69  ec = forward.stop(reproc::cleanup::wait, reproc::infinite, &exit_status);
70  if (ec) { return fail(ec); }
71 
72  return static_cast<int>(exit_status);
73 }
