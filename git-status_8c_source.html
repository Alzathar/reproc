28  if (error == REPROC_FILE_NOT_FOUND) {

35  }

39  // show how reproc_close works.

41

44  char *output = malloc(1);
45  if (!output) { goto cleanup; }
46  output[0] = '\0';

50  // Read the entire output of the child process. I've found this pattern to be
51  // the most readable when reading the entire output of a child process. The
52  // while loop keeps running until an error occurs in reproc_read (the child
53  // process closing its output stream is also reported as an error).
54  while (true) {

59

62  if (!realloc_result) {

65  }
66  output = realloc_result;
67 
68  memcpy(output + output_length, buffer, bytes_read);
69  output_length += bytes_read;
70  }
71 
72  // Check that the while loop stopped because the output stream of the child
73  // process was closed and not because of another error
74  if (error != REPROC_STREAM_CLOSED) {
75  free(output);
76  goto cleanup;
77  }
78

87  // the parent process explicitly waits for it after it has exited.
